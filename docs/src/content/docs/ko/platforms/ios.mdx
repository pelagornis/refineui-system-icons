# iOS 가이드

iOS 네이티브 앱에서 RefineUI System Icons를 사용하는 방법을 안내합니다.

## 설치

### Swift Package Manager

Xcode에서 File → Add Package Dependencies를 선택하고 다음 URL을 추가하세요:

```
https://github.com/pelagornis/refineui-system-icons
```

## 기본 사용법

### Import

```swift
import RefineIcons
```

### UIImage로 사용

```swift
// Filled 스타일 아이콘
let addIcon = UIImage(systemName: "add.24.filled")

// Regular 스타일 아이콘
let addIconRegular = UIImage(systemName: "add.24.regular")

// UIImageView에 설정
let imageView = UIImageView(image: addIcon)
```

## 실제 사용 예제

### UIButton에서 사용

```swift
import UIKit
import RefineIcons

class ViewController: UIViewController {
    override func viewDidLoad() {
        super.viewDidLoad()
        
        // 버튼 생성
        let addButton = UIButton(type: .system)
        addButton.setImage(UIImage(systemName: "add.24.filled"), for: .normal)
        addButton.setTitle("추가", for: .normal)
        addButton.addTarget(self, action: #selector(addButtonTapped), for: .touchUpInside)
        
        // 레이아웃 설정
        addButton.translatesAutoresizingMaskIntoConstraints = false
        view.addSubview(addButton)
        
        NSLayoutConstraint.activate([
            addButton.centerXAnchor.constraint(equalTo: view.centerXAnchor),
            addButton.centerYAnchor.constraint(equalTo: view.centerYAnchor)
        ])
    }
    
    @objc func addButtonTapped() {
        print("추가 버튼 클릭")
    }
}
```

### UINavigationBar에서 사용

```swift
import UIKit
import RefineIcons

class ViewController: UIViewController {
    override func viewDidLoad() {
        super.viewDidLoad()
        
        // 네비게이션 바 설정
        navigationItem.title = "My App"
        
        // 오른쪽 버튼 추가
        let addButton = UIBarButtonItem(
            image: UIImage(systemName: "add.24.filled"),
            style: .plain,
            target: self,
            action: #selector(addButtonTapped)
        )
        
        navigationItem.rightBarButtonItem = addButton
    }
    
    @objc func addButtonTapped() {
        print("추가 버튼 클릭")
    }
}
```
